27  package com.netbout.client;
28  
29  import com.google.common.base.Function;
30  import com.google.common.collect.Lists;
31  import com.jcabi.aspects.Loggable;
32  import com.jcabi.http.Request;
33  import com.jcabi.http.response.RestResponse;
34  import com.jcabi.http.response.XmlResponse;
35  import com.jcabi.xml.XML;
36  import com.netbout.spi.Message;
37  import java.io.IOException;
38  import java.net.HttpURLConnection;
39  import java.text.ParseException;
40  import java.util.Date;
41  import java.util.Iterator;
42  import java.util.LinkedList;
43  import java.util.NoSuchElementException;
44  import java.util.Queue;
45  import lombok.EqualsAndHashCode;
46  import lombok.ToString;
47  import org.apache.commons.lang3.time.DateFormatUtils;
48  
49  /**
50   * REST bout iterator.
51   *
52   * @author Yegor Bugayenko (yegor@teamed.io)
53   * @version $Id: 0cae01924ed2c7e5c87ab90144c11bd500521125 $
54   * @since 2.14
55   */
56  @ToString
57  @Loggable(Loggable.DEBUG)
58  @EqualsAndHashCode(of = { "request", "messages" })
59  final class RtMessageIterator implements Iterator<Message> {
60  
61      /**
62       * Pre-fetched messages.
63       */
64      private final transient Queue<Message> messages;
65  
66      /**
67       * Request to use.
68       */
69      private transient Request request;
70  
71      /**
72       * Has more?
73       */
74      private transient boolean more;
75  
76      /**
77       * Public ctor.
78       * @param req Request to use
79       */
80      RtMessageIterator(final Request req) {
81          this.request = req;
82          this.messages = new LinkedList<Message>();
83          this.more = true;
84      }
85  
86      @Override
87      public boolean hasNext() {
88          if (this.messages.isEmpty() && this.more) {
89              try {
90                  this.fetch();
91              } catch (final IOException ex) {
92                  throw new IllegalStateException(ex);
93              }
94          }
95          return !this.messages.isEmpty();
96      }
97  
98      @Override
99      public Message next() {
100         if (!this.hasNext()) {
101             throw new NoSuchElementException("end of the bout");
102         }
103         return this.messages.poll();
104     }
105 
106     @Override
107     public void remove() {
108         throw new UnsupportedOperationException("#remove()");
109     }
110 
111     /**
112      * Fetch more.
113      * @throws IOException If fails
114      */
115     private void fetch() throws IOException {
116         final XmlResponse response = this.request.fetch()
117             .as(RestResponse.class)
118             .assertStatus(HttpURLConnection.HTTP_OK)
119             .as(XmlResponse.class);
120         final XML xml = response.xml();
121         this.messages.addAll(
122             Lists.transform(
123                 xml.nodes("/page/bout/messages/message"),
124                 new Function<XML, Message>() {
125                     @Override
126                     public Message apply(final XML node) {
127                         return RtMessageIterator.msg(node);
128                     }
129                 }
130             )
131         );
132         if (xml.nodes("/page/bout/messages/message ").isEmpty()) {
133             this.more = false;
134         } else {
135             this.request = response.rel(
136                 // @checkstyle LineLength (1 line)
137                 "/page/bout/messages/message[last()]/links/link[@rel='more']/@href"
138             );
139         }
140     }
141 
142     /**
143      * Turn XML into a message.
144      * @param xml The XML
145      * @return Message
146      */
147     private static Message msg(final XML xml) {
148         // @checkstyle AnonInnerLengthCheck (50 lines)
149         return new Message() {
150             @Override
151             public long number() {
152                 return Long.parseLong(
153                     xml.xpath("number/text()").get(0)
154                 );
155             }
156             @Override
157             public Date date() throws IOException {
158                 try {
159                     return DateFormatUtils.ISO_DATETIME_FORMAT.parse(
160                         xml.xpath("date/text()").get(0)
161                     );
162                 } catch (final ParseException ex) {
163                     throw new IOException(ex);
164                 }
165             }
166             @Override
167             public String text() {
168                 return xml.xpath("text/text()").get(0);
169             }
170             @Override
171             public String author() {
172                 return xml.xpath("author/text()").get(0);
173             }
174         };
175     }

177 }

79      @Override
80      public Response act(final Request req) throws IOException {
81          final String invite = new RqForm.Smart(
82              new RqForm.Base(req)
83          ).single("name");
84          final Bout bout = new RqBout(this.base, req).bout();
85          final String guest;
86          if (MAIL_MASK.matcher(invite).find()) {
87              guest = this.inviteByEmail(invite, bout);
88          } else {
89              guest = invite;
90          }
91          try {
92              bout.friends().invite(guest);
93          } catch (
94              final Friends.UnknownAliasException | IllegalArgumentException ex
95          ) {
96              throw new RsFailure(ex);
97          }
98          throw new RsForward(
99              new RsFlash(
100                 String.format(
101                     "\"%s\" invited to the bout #%d",
102                     guest, bout.number()
103                 ),
104                 Level.INFO
105             )
106         );
107     }

109     /**
110      * Invite a user to participate on NetBout by email.
111      * @param invite Email
112      * @param bout Bout
113      * @return Alias.
114      * @throws IOException If fails.
115      */
116     public String inviteByEmail(@NotNull(message = "Invite can't be NULL")
117         final String invite, final Bout bout) throws IOException {
118         // @checkstyle MultipleStringLiteralsCheck (1 line)
119         final String alias = invite.replace("@", "-").replace(".", "-");
120         final String urn = String.format(
121             "urn:email:%s",
122             DigestUtils.sha1Hex(alias)
123         );
124         this.base.user(URN.create(urn)).aliases().add(alias);
125         try {
126             new RqAlias(this.base, new RqWithAuth(urn)).alias()
127                 .email(invite, urn, bout);
128         } catch (final IOException ex) {
129             throw new RsFailure(ex);
130         }
131         return alias;
132     }

32  /**
33   * Searchable.
34   *
35   * @author Ivan Inozemtsev (ivan.inozemtsev@gmail.com)
36   * @version $Id: ed0ba1bcc46e2e6be1a9014506028aff0db431da $
37   * @since 2.15.2
38   * @param <T> The type of items
39   */
40  @Immutable
41  public interface Searchable<T> {
42      /**
43       * Find items matching the search term.
44       * @param term String to search
45       * @return Items matching the search term
46       * @throws IOException If fails
47       */

50  }

27  package com.netbout.email;
28  
29  import org.xembly.Directives;
30  import org.xembly.ImpossibleModificationException;
31  import org.xembly.Xembler;
32  
33  /**
34   * Integrates with Gmail's View Actions.
35   * @author Matteo Barbieri (barbieri.matteo@gmail.com)
36   * @version $Id: a2056445ddd27ecbc57ecb905d3151eec0d845ee $
37   * @since 2.15
38   */
39  final class GmailViewAction {
40  
41      /**
42       * The bout number.
43       */
44      private final transient long number;
45  
46      /**
47       * Primary ctor.
48       * @param nbr The bout number
49       */
50      GmailViewAction(final long nbr) {
51          this.number = nbr;
52      }
53  
54      /**
55       * Returns the XML section for Gmail ViewAction.
56       * @return The XML
57       * @checkstyle MultipleStringLiteralsCheck (45 lines)
58       */
59      @SuppressWarnings("PMD.AvoidDuplicateLiterals")
60      public String xml() {
61          try {
62              return new Xembler(
63                  new Directives()
64                      .add("div")
65                      .attr("itemscope", "")
66                      .attr("itemtype", "http://schema.org/EmailMessage")
67                      .add("meta")
68                      .attr("itemprop", "description")
69                      .attr("content", "View Bout").up()
70                      .add("div")
71                      .attr("itemprop", "potentialAction")
72                      .attr("itemscope", "")
73                      .attr("itemtype", "http://schema.org/ViewAction")
74                      .add("link").attr("itemprop", "target")
75                      .attr(
76                          "href",
77                          String.format(
78                              "http://www.netbout.com/b/%d", this.number
79                          )
80                      ).up()
81                      .add("link").attr("itemprop", "url")
82                      .attr(
83                          "href",
84                          String.format(
85                              "http://www.netbout.com/b/%d", this.number
86                          )
87                      ).up()
88                      .add("meta")
89                      .attr("itemprop", "name")
90                      .attr("content", "View Bout").up().up()
91                      .add("div")
92                      .attr("itemprop", "publisher")
93                      .attr("itemscope", "")
94                      .attr("itemtype", "http://schema.org/Organization")
95                      .add("meta")
96                      .attr("itemprop", "name")
97                      .attr("content", "Netbout").up()
98                      .add("link")
99                      .attr("itemprop", "url")
100                     .attr("href", "http://www.netbout.com")
101             ).xml();
102         } catch (final ImpossibleModificationException ex) {
103             throw new IllegalStateException(ex);
104         }
105     }
106 }

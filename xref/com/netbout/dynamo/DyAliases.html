27  package com.netbout.dynamo;
28  
29  import com.amazonaws.services.dynamodbv2.model.Select;
30  import com.google.common.base.Function;
31  import com.google.common.collect.Iterables;
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.aspects.Loggable;
34  import com.jcabi.aspects.Tv;
35  import com.jcabi.dynamo.Attributes;
36  import com.jcabi.dynamo.Conditions;
37  import com.jcabi.dynamo.Item;
38  import com.jcabi.dynamo.QueryValve;
39  import com.jcabi.dynamo.Region;
40  import com.jcabi.log.Logger;
41  import com.jcabi.urn.URN;
42  import com.netbout.spi.Alias;
43  import com.netbout.spi.Aliases;
44  import java.io.IOException;
45  import java.util.Locale;
46  import lombok.EqualsAndHashCode;
47  import lombok.ToString;
48  
49  /**
50   * Dynamo Aliases.
51   *
52   * @author Yegor Bugayenko (yegor@teamed.io)
53   * @version $Id: 2a2578c6654106d24c65b0086e06612c0b1107df $
54   * @since 2.0
55   */
56  @Immutable
57  @Loggable(Loggable.DEBUG)
58  @ToString(of = "urn")
59  @EqualsAndHashCode(of = { "region", "urn" })
60  final class DyAliases implements Aliases {
61  
62      /**
63       * Table name.
64       */
65      public static final String TBL = "aliases";
66  
67      /**
68       * HASH attribute.
69       */
70      public static final String HASH = "alias";
71  
72      /**
73       * URN attribute.
74       */
75      public static final String ATTR_URN = "urn";
76  
77      /**
78       * Locale attribute.
79       */
80      public static final String ATTR_LOCALE = "locale";
81  
82      /**
83       * Photo attribute.
84       */
85      public static final String ATTR_PHOTO = "photo";
86  
87      /**
88       * Photo attribute.
89       * @since 2.12
90       */
91      public static final String ATTR_EMAIL = "email";
92  
93      /**
94       * Index name.
95       */
96      private static final String INDEX = "users";
97  
98      /**
99       * Region to work with.
100      */
101     private final transient Region region;
102 
103     /**
104      * URN of the user.
105      */
106     private final transient URN urn;
107 
108     /**
109      * Ctor.
110      * @param reg Region we're in
111      * @param user URN of the user
112      */
113     DyAliases(final Region reg, final URN user) {
114         this.region = reg;
115         this.urn = user;
116     }
117 
118     @Override
119     public String check(final String name) {
120         final String answer;
121         if (name.length() < Tv.FOUR) {
122             answer = "too short, must be 4 letters at least";
123         } else if (name.length() > Tv.TWENTY) {
124             answer = "too long, must be 20 letters at most";
125         } else if (name.matches("[a-z0-9]+")) {
126             if (new Everybody(this.region).occupied(name)) {
127                 answer = "this alias is occupied";
128             } else {
129                 answer = "";
130             }
131         } else {
132             answer = "only lower case English letters and numbers are accepted";
133         }
134         return answer;
135     }
136 
137     @Override
138     public Alias add(final String name) throws IOException {
139         if (name.isEmpty()) {
140             throw new IllegalArgumentException("alias can't be empty");
141         }
142         if (name.length() > Tv.HUNDRED) {
143             throw new IllegalArgumentException("alias is too long");
144         }
145         if (new Everybody(this.region).occupied(name)) {
146             throw new IllegalArgumentException(
147                 String.format("alias '%s' is occupied", name)
148             );
149         }
150         final Item item = this.region.table(DyAliases.TBL).put(
151             new Attributes()
152                 .with(DyAliases.ATTR_URN, this.urn)
153                 .with(DyAliases.HASH, name)
154                 .with(DyAliases.ATTR_PHOTO, Alias.BLANK)
155                 .with(DyAliases.ATTR_LOCALE, Locale.ENGLISH)
156         );
157         Logger.info(this, "alias @%s added to %s", name, this.urn);
158         return new DyAlias(this.region, item);
159     }
160 
161     @Override
162     public Iterable<Alias> iterate() {
163         return Iterables.transform(
164             this.region
165                 .table(DyAliases.TBL)
166                 .frame()
167                 .where(DyAliases.ATTR_URN, Conditions.equalTo(this.urn))
168                 .through(
169                     new QueryValve()
170                         .withIndexName(DyAliases.INDEX)
171                         .withConsistentRead(false)
172                         .withSelect(Select.ALL_PROJECTED_ATTRIBUTES)
173                 ),
174             new Function<Item, Alias>() {
175                 @Override
176                 public Alias apply(final Item item) {
177                     return new DyAlias(DyAliases.this.region, item);
178                 }
179             }
180         );
181     }
182 
183 }

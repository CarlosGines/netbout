27  package com.netbout.rest.bout;
28  
29  import com.netbout.mock.MkBase;
30  import com.netbout.spi.Message;
31  import java.util.Date;
32  import org.hamcrest.MatcherAssert;
33  import org.hamcrest.Matchers;
34  import org.junit.Test;
35  import org.xembly.Xembler;
36  
37  /**
38   * Test case for {@link XeMessage}.
39   * @author Dmitry Zaytsev (dmitry.zaytsev@gmail.com)
40   * @version $Id: 78c7602154f4ae758269bc71a5d2d2f8c40990cb $
41   * @since 2.16.1
42   */
43  public final class XeMessageTest {
44      /**
45       * XeMessage can handle non printable characters.
46       * @throws Exception If there is some problem inside
47       */
48      @Test
49      public void handlesNonPrintableChars() throws Exception {
50          final String xml = new Xembler(
51              new XeMessage(
52                  new MkBase().randomBout(),
53                  new Message() {
54                      @Override
55                      public long number() {
56                          return 0L;
57                      }
58                      @Override
59                      public Date date() {
60                          return new Date();
61                      }
62                      @Override
63                      public String text() {
64                          return String.format("text contains %c", 0x0);
65                      }
66                      @Override
67                      public String author() {
68                          return String.format("author  contains %c", 0x0);
69                      }
70                  }
71              ).toXembly()
72          )
73              .xml();
74          MatcherAssert.assertThat(
75              xml,
76              Matchers.containsString("<author>author  contains \\u0000</author>")
77          );
78          MatcherAssert.assertThat(
79              xml,
80              Matchers.containsString("<text>text contains \\u0000</text>")
81          );
82      }
83  }
